trim ​

trim(str)

去除字符串两端的空格

javascript
const str = ' 你好 '
console.log(str)                   // 输出“ 你好 ”
console.log(stringC.trim(str))     // 输出 “你好”

replaceAll ​

替换字符串中的某个字符或子串

replaceAll(str, searchValue, replaceValue)

参数：

  1. str
  2. searchValue 被查找的字符或子串
  3. replaceValue 替换的字符或子串
javascript
const str = "这是一段文本，被替换的词是ABC"
console.log(stringC.replaceAll(str, 'ABC', 'ABCDEF'))// 输出“这是一段文本，被替换的词是ABCDEF”

getSearchParamsValue ​

获取查询字符串 key的值

getSearchParamsValue(url, key)

参数：

  1. url
  2. key
javascript
const url = "http://localhost:8088?id=123&name='张三'"
const key = 'name'
console.log(textC.getSearchParamsValue(url, key)) // 输出张三

setSearchParamsValue ​

修改查询字符串的某个值

setSearchParamsValue(url, key, value)

参数：

  1. url
  2. key 被替换的key值
  3. value 替换后的值
javascript
const url = "http://localhost:8088?id=123&name='张三'"
const key = 'name'
console.log(textC.setSearchParamsValue(url, key),'李四') // http://localhost:8088?id=123&name='李四'

queryStringToObject ​

url转对象

javascript
const query1 = "http://localhost:8088?name='张三'&id=2&year=18"
const query2 = "?aaa='a'&bbb=b&ccc=c"
const query3 = "aaa='a'&bbb=b&ccc=c"
console.log(111, stringC.queryStringToObject(query1)) // {name: '张三', id: 2, year: 18}
console.log(222, stringC.queryStringToObject(query2)) // {aaa: 'a', bbb: 'b', ccc: 'c'}
console.log(333, stringC.queryStringToObject(query3)) // {aaa: 'a', bbb: 'b', ccc: 'c'}

objectToQueryString ​

对象转url

javascript
const obj = {
  aaa: '1',
  bbb: '2',
  ccc: '3'
}
console.log(111, stringC.objectToQueryString(obj)) // 'aaa=1&bbb=2&ccc=3'
