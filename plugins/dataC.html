cloneDeep ​

深拷贝

cloneDeep(arr)

javascript
const arr = [1, 2, 3, 4, 5]
const arr2 = arr
const arr3 = dataC.cloneDeep(arr)
arr2.splice(arr2.length - 1, 1) // 删除arr2数组最后一个元素
console.log(arr)   // 输出[1, 2, 3, 4]
console.log(arr2)  // 输出[1, 2, 3, 4]
console.log(arr3)  // 输出[1, 2, 3, 4,5]

getItemByValue ​

匹配数组中特定value对应的值

getItemByValue(arr, value, valueKey)

参数：

  1. arr被匹配的数组
  2. value 需要匹配value的值
  3. valueKey 需要匹配value的key值，默认值为'value'
javascript
const arr1 = [
  {value: 1, name: '标签1', code: 'code1'},
  {value: 2, name: '标签2', code: 'code2'},
  {value: 3, name: '标签3', code: 'code3'}
]
console.log(dataC.getItemByValue(arr1, 2)) // 输出{value: 2, name: '标签2', code: 'code2'}
javascript
const arr = [
  {id: 1, name: '标签1', code: 'code1'},
  {id: 2, name: '标签2', code: 'code2'},
  {id: 3, name: '标签3', code: 'code3'}
]
console.log(dataC.getItemByValue(arr2, 2, 'id')) // 输出{id: 2, name: '标签2', code: 'code2'}

filterArray ​

元素为对象格式的数组去重

filterArray(arr, key, key2)

参数：

  1. arr arr需要去重的对象数组

  2. key 去重的标识，即将数组中key重复的数据过滤掉,只保留一个

  3. key2第二个去重的标识 即将数组中key以及key2都重复的数据过滤掉，只保留一个

javascript
const arr1 = [
  {value: 1, name: '标签1', code: 'code1'},
  {value: 1, name: '标签2', code: 'code2'},
  {value: 3, name: '标签3', code: 'code3'}
]
const result1 = dataC.filterArray(arr1, 'value')  // 第二条数据被过滤 因为value相等
const result2 = dataC.filterArray(arr1, 'value', 'code') // 第二条数据没有被过滤 因为value和code不同时相等

isEmpty ​

数据是否为空

isEmpty(value)

返回true的情况如下 数字0返回false

  1. undefind

  2. null

  3. ''

  4. 数组为空[]

  5. 对象为空{}

javascript
console.log(isEmpty(undefined))  // 输出true
console.log(isEmpty(null))       // 输出true
console.log(isEmpty(''))         // 输出true
console.log(isEmpty([]))         // 输出true
console.log(isEmpty({}))         // 输出true
console.log(isEmpty(0))          // 输出false

safeObject ​

判断是否为{"a": 1, "b": 2} 或 {a: 1, b: 2}这种可转成对象的格式，即是否可通过JOSN.parse转成对象,如果是转成对象并返回，否则返回null 日常开发中对str进行转换时可能str并是json字符串会导致报错，可以用这个方法进行转换

javascript
const a = '{"a": 1, "b": 2}'
console.log(11111, dataC.safeObject(a))    // 输出 {a: 1, b: 2}
const b = '{a: 1, b: 2}'   
console.log(22222, dataC.safeObject(b))    // 输出 {a: 1, b: 2}
const c = 'a: 1, b: 2'
console.log(33333, dataC.safeObject(c))    // 输出 null
